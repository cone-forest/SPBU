\begin{problem}{102}
Вычислить $(x-1-i)(x-1+i)(x+1+i)(x+1-i)$.
\end{problem}
\begin{solution}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Ваше решение задачи здесь

$
(x - (1 + i))(x - (1 - i))(x + (1 + i))(x + (1 - i)) = \\
(x - (1 + i))(x + (1 + i))(x + (1 - i))(x - (1 - i)) = \\
(x^2 - (1 + i)^2)(x^2 - (1 - i)^2) = \\
(x^2 - 2i)(x^2 + 2i) = \\
(x^4 + 4) \\
$
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\end{solution}
